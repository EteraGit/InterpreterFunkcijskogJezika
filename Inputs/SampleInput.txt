add(x,0):=x
add(x,Sc(y)):=Sc(add(x,y))

pd(0):=0
pd(Sc(y)):=y

sub(x,0):=x
sub(x,Sc(y)):=pd(sub(x,y))

// zakomentirano za sad (x+y) ::= add(x,y)   // nice to have

f(x,y):=add(Sc(Sc(x)),y)

g(x,y):=add(sub(x,y), y)

h(x,y,z):=mul(Sc(z), sub(x,y))

mul(x,0):=0
mul(x,Sc(y)):=add(x,mul(x,y))

factorial(0):=1
factorial(Sc(n)):=mul(Sc(n),factorial(n))

[Positive](0):=0
[Positive](Sc(x)):=1

{Greater}(x,y):<=>{Positive}(sub(x,y))  // za x = y ne radi kako treba

div(x,y):=pd((mu z){Greater}(mul(z,y),x))

mod(x,y):=sub(x,mul(y,div(x,y)))   // (x-(y*(x//y)))

{Equal}(x,y):<=>!{Greater}(x,y) && !{Greater}(y,x)

{Divides}(x,y):<=>{Equal}(mod(y,x),Z(x))  // moze i Equal(mod(x,y),0)

{Prime}(p):<=>{Equal}((#d<=p){Divides}(d,p), 2)

nextprime(p):=(mu q)({Prime}(q) && {Greater}(q,p))  // eventualno AND(...)
